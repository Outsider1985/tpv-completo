SELECT
							ROUND( SUM( precio_total ), 2 ) AS total,
							(
							SELECT
								ROUND( AVG( suma_total ), 2 ) AS promedio_dia 
							FROM
								(
								SELECT
									SUM( precio_total ) AS suma_total,
									DAYNAME( fecha_emision ) AS fecha 
								FROM
									ventas 
								WHERE
									DAYNAME( fecha_emision ) = DAYNAME( '2022-06-14' ) 
									AND activo = 1
								GROUP BY
									fecha_emision 
								) media 
							GROUP BY
								fecha 
							) AS media 
						FROM
							`ventas` 
						WHERE
							fecha_emision = '2022-06-14' 
							AND ventas.activo = 1
